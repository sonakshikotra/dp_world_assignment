--- Create table for AIS data
CREATE TABLE ais_2020_01 (
mmsi VARCHAR(20),
basedatetime TIMESTAMP,
LAT DECIMAL,
LON DECIMAL,
SOG VARCHAR(50),
COG VARCHAR(50),
Heading VARCHAR(50),
VesselName VARCHAR(50),
IMO  VARCHAR(50),
CallSign  VARCHAR(50),
VesselType INTEGER,
Status VARCHAR(50),
Length DECIMAL,
Width DECIMAL,
Draft DECIMAL,
Cargo VARCHAR(50),
TransceiverClass VARCHAR(50)
);
 
--- Create table for port coordinates containing only necessary columns
CREATE TABLE port (
“Main Port Name” VARCHAR(200),
“UN/LOCODE” VARCHAR(200),
“Country Code” VARCHAR(200),
Latitude DECIMAL,
Longitude DECIMAL
);
